    "Modifiers"

        "modifier_spawn_undead"
        {
            "Passive"           "1"
            "IsHidden"          "1"

            "OnCreated"
            {
                "RunScript"
                {
                    "ScriptFile"    "abilities/garage.lua"

                    "Value"         "45"
                }
            }

            "ThinkInterval"     "7.0"
            "OnIntervalThink"
            {
                "RunScript"
                {
                    "ScriptFile"    "abilities/spawn.lua"
                    "Function"      "SpawnUnit"
                    "UnitName"      "undead"
                }
            }
        }
    }
}
